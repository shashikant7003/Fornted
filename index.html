// 🛑 Apne Render ka real URL yahan check karo (NO extra spaces, NO slash at the end) const BACKEND_URL = "https://downloader-bbcs.onrender.com"; async function fetchLink() { const urlInput = document.getElementById('video-url').value; const btn = document.getElementById('dl-btn'); const loader = document.getElementById('loader'); const errorBox = document.getElementById('error-display'); if(!urlInput) return alert("Please paste a link first!"); btn.style.opacity = '0.5'; btn.innerText = 'Connecting...'; loader.style.display = 'flex'; errorBox.style.display = 'none'; try { const response = await fetch(`${BACKEND_URL}/api/get-download-link`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ url: urlInput }) }); // Agar Render ne 200 OK nahi diya, toh pehle uski wajah pakdo if (!response.ok) { const rawText = await response.text(); throw new Error(`Server Response (${response.status}): ${rawText.substring(0, 100)}`); } const result = await response.json(); if(result.success && result.download_url) { window.location.href = result.download_url; } else { throw new Error(result.error || "Failed to fetch stream."); } } catch (err) { // Yeh line ab asli debugging karegi errorBox.innerText = "🚨 " + err.message; errorBox.style.display = 'block'; } finally { btn.style.opacity = '1'; btn.innerText = 'Download Now'; loader.style.display = 'none'; } }
